  # Suspend
  kubectl patch cronjob {{ include "cronjob.fullname" . }} -n {{ .Release.Namespace }} -p '{"spec":{"suspend":true}}'

  # Resume
  kubectl patch cronjob {{ include "cronjob.fullname" . }} -n {{ .Release.Namespace }} -p '{"spec":{"suspend":false}}'
