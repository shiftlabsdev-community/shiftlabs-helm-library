1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "statefulset.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "statefulset.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "statefulset.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ index .Values.service.ports 0 "port" }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "statefulset.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Get pod status:
  kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "statefulset.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

3. View logs:
  kubectl logs --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "statefulset.name" . }},app.kubernetes.io/instance={{ .Release.Name }}"

4. StatefulSet info:
  - Replicas: {{ .Values.replicaCount }}
  - Pod Management: {{ .Values.statefulset.podManagementPolicy }}
  - Update Strategy: {{ .Values.statefulset.updateStrategy.type }}
  kubectl get statefulset --namespace {{ .Release.Namespace }} {{ include "statefulset.fullname" . }}

{{- if .Values.persistence.enabled }}
5. Persistent Volume Claims:
  kubectl get pvc --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "statefulset.name" . }}"
{{- end }}

{{- if .Values.ExternalSecret.enabled }}
6. External Secret is enabled ({{ .Values.ExternalSecret.type }}):
  - Secret store: {{ .Values.ExternalSecret.name }}
  - Verify secret: kubectl get secret {{ include "statefulset.fullname" . }} -n {{ .Release.Namespace }}
{{- end }}

{{- if .Values.networkPolicy.enabled }}
7. NetworkPolicy is enabled:
  - Check network policies: kubectl get networkpolicy {{ include "statefulset.fullname" . }} -n {{ .Release.Namespace }}
{{- end }}
